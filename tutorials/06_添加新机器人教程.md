# 教程六：添加新机器人教程

## 6.1 概述

在本教程中，我们将学习如何向 LeggedGym-Ex 添加一个全新的机器人。我们将以一个虚构的四足机器人 "MyRobot" 为例，逐步完成整个过程。

## 6.2 准备工作

### 检查清单

在开始之前，请确保你有：

- [ ] 机器人的 URDF 文件
- [ ] 机器人的 3D 网格文件（.obj 或 .stl）
- [ ] 了解机器人的关节顺序和名称
- [ ] 确定机器人的默认站立姿势

### 文件结构规划

我们将创建以下文件：

```
legged_gym/envs/myrobot/
├── __init__.py
├── myrobot.py              # 环境类
└── myrobot_config.py       # 配置类

resources/robots/myrobot/
├── urdf/
│   └── myrobot.urdf
└── meshes/
    ├── base.obj
    ├── thigh.obj
    └── ...
```

## 6.3 步骤一：添加机器人资产

### 6.3.1 创建资产目录

## 6.4 步骤二：创建配置类

### 6.4.1 创建配置文件

创建 `legged_gym/envs/myrobot/myrobot_config.py`：

```python
from legged_gym import *
from legged_gym.envs.base.legged_robot_config import LeggedRobotCfg, LeggedRobotCfgPPO

class MyRobotCfg(LeggedRobotCfg):
    """MyRobot 环境配置"""

    class env(LeggedRobotCfg.env):
        num_envs = 4096
        num_observations = 45  # 根据你的观测调整
        num_actions = 12        # 根据你的关节数量调整
        episode_length_s = 20

    class terrain(LeggedRobotCfg.terrain):
        mesh_type = 'plane'     # 从平面开始
        curriculum = False

    class init_state(LeggedRobotCfg.init_state):
        pos = [0.0, 0.0, 0.4]   # 调整初始高度

        # 设置默认关节角度（根据你的机器人）
        default_joint_angles = {
            "FL_hip_joint": 0.1,
            "FL_thigh_joint": -0.8,
            "FL_calf_joint": 1.5,
            "FR_hip_joint": 0.1,
            "FR_thigh_joint": -0.8,
            "FR_calf_joint": 1.5,
            "RL_hip_joint": 0.1,
            "RL_thigh_joint": -0.8,
            "RL_calf_joint": 1.5,
            "RR_hip_joint": 0.1,
            "RR_thigh_joint": -0.8,
            "RR_calf_joint": 1.5,
        }

    class control(LeggedRobotCfg.control):
        # PD 控制器参数
        stiffness = {
            "joint": 20.0,    # 调整刚度
        }
        damping = {
            "joint": 0.5,      # 调整阻尼
        }
        action_scale = 0.25
        decimation = 4

    class asset(LeggedRobotCfg.asset):
        # URDF 文件路径
        file = '{LEGGED_GYM_ROOT_DIR}/resources/robots/myrobot/urdf/myrobot.urdf'
        name = "myrobot"

        # 调整这些参数
        foot_name = "foot"       # 脚链接的名称
        terminate_after_contacts_on = ["base"]  # 接触这些链接时终止
        self_collisions = 0      # 1 启用自碰撞，0 禁用
        flip_visual_attachments = False

    class rewards(LeggedRobotCfg.rewards):
        soft_dof_pos_limit = 0.9
        base_height_target = 0.4  # 目标基座高度

        class scales(LeggedRobotCfg.rewards.scales):
            # 命令跟踪
            tracking_lin_vel = 1.0
            tracking_ang_vel = 0.5

            # 平滑性
            lin_vel_z = -0.5
            base_height = -2.0
            ang_vel_xy = -0.05
            orientation = -1.0
            dof_vel = -5.e-4
            action_rate = -0.01

            # 限制
            dof_pos_limits = -1.0
            collision = -1.0

            # 步态
            feet_air_time = 1.0

    class commands(LeggedRobotCfg.commands):
        curriculum = False
        num_commands = 4

        class ranges(LeggedRobotCfg.commands.ranges):
            lin_vel_x = [-0.5, 0.5]
            lin_vel_y = [-0.5, 0.5]
            ang_vel_yaw = [-1.0, 1.0]
            heading = [-3.14, 3.14]

    class domain_rand(LeggedRobotCfg.domain_rand):
        # 初始阶段可以禁用领域随机化
        randomize_friction = False
        randomize_base_mass = False
        push_robots = False

class MyRobotCfgPPO(LeggedRobotCfgPPO):
    """MyRobot 训练配置"""

    class runner(LeggedRobotCfgPPO.runner):
        run_name = 'myrobot_simple'
        if SIMULATOR == "genesis":
            run_name += '_genesis'
        elif SIMULATOR == "isaacgym":
            run_name += '_isaacgym'
        experiment_name = 'myrobot'
        save_interval = 200
        max_iterations = 1500
```

### 关键点说明

1. **关节顺序很重要**：`default_joint_angles` 中的顺序必须与 URDF 中的顺序一致
2. **初始高度**：根据机器人大小调整 `pos[2]` 和 `base_height_target`
3. **脚名称**：`foot_name` 必须与 URDF 中的脚链接名称匹配
4. **接触终止**：`terminate_after_contacts_on` 列出不应该接触地面的链接

## 6.5 步骤三：创建环境类

### 6.5.1 创建环境文件

创建 `legged_gym/envs/myrobot/myrobot.py`：

```python
from legged_gym import *
import torch

from legged_gym.envs.base.legged_robot import LeggedRobot
from legged_gym.utils.math_utils import torch_rand_float
from .myrobot_config import MyRobotCfg

class MyRobot(LeggedRobot):
    """MyRobot 环境类"""

    def _reset_dofs(self, env_ids):
        """重置关节状态"""

        dof_pos = torch.zeros((len(env_ids), self.num_actions),
                              dtype=torch.float, device=self.device, requires_grad=False)
        dof_vel = torch.zeros_like(dof_pos)

        # 添加小随机扰动
        # 假设关节 0, 3, 6, 9 是髋关节
        dof_pos[:, [0, 3, 6, 9]] = self.simulator.default_dof_pos[:, [0, 3, 6, 9]] + \
            torch_rand_float(-0.2, 0.2, (len(env_ids), 4), self.device)

        # 其他关节添加更大的扰动
        dof_pos[:, [1, 2, 4, 5, 7, 8, 10, 11]] = self.simulator.default_dof_pos[:, [1, 2, 4, 5, 7, 8, 10, 11]] + \
            torch_rand_float(-0.3, 0.3, (len(env_ids), 8), self.device)

        self.simulator.reset_dofs(env_ids, dof_pos, dof_vel)

    def compute_observations(self):
        """计算观测值"""

        self.obs_buf = torch.cat((
            self.commands[:, :3] * self.commands_scale,
            self.simulator.projected_gravity,
            self.simulator.base_ang_vel * self.obs_scales.ang_vel,
            (self.simulator.dof_pos - self.simulator.default_dof_pos) * self.obs_scales.dof_pos,
            self.simulator.dof_vel * self.obs_scales.dof_vel,
            self.actions
        ), dim=-1)

        # 添加噪声
        if self.add_noise:
            self.obs_buf += (2 * torch.rand_like(self.obs_buf) - 1) * self.noise_scale_vec

        # 特权观测（如果需要）
        if self.num_privileged_obs is not None:
            self.privileged_obs_buf = torch.cat(
                (
                    self.simulator.base_lin_vel * self.obs_scales.lin_vel,
                    self.simulator.base_ang_vel * self.obs_scales.ang_vel,
                    self.simulator.projected_gravity,
                    self.commands[:, :3] * self.commands_scale,
                    (self.simulator.dof_pos - self.simulator.default_dof_pos) * self.obs_scales.dof_pos,
                    self.simulator.dof_vel * self.obs_scales.dof_vel,
                    self.actions,
                    self.last_actions,
                ),
                dim=-1,
            )

    def _get_noise_scale_vec(self):
        """获取噪声缩放向量"""

        noise_vec = torch.zeros_like(self.obs_buf[0])
        self.add_noise = self.cfg.noise.add_noise
        noise_scales = self.cfg.noise.noise_scales
        noise_level = self.cfg.noise.noise_level

        # 调整这些索引以匹配你的观测
        noise_vec[:3] = 0.  # 命令
        noise_vec[3:6] = noise_scales.gravity * noise_level
        noise_vec[6:9] = noise_scales.ang_vel * noise_level * self.obs_scales.ang_vel
        noise_vec[9:21] = noise_scales.dof_pos * noise_level * self.obs_scales.dof_pos
        noise_vec[21:33] = noise_scales.dof_vel * noise_level * self.obs_scales.dof_vel
        noise_vec[33:45] = 0.

        return noise_vec

    def _reset_root_states(self, env_ids):
        """重置基座状态"""

        if self.simulator.custom_origins:
            base_pos = self.simulator.base_init_pos.reshape(1, -1).repeat(len(env_ids), 1)
            base_pos += self.simulator.env_origins[env_ids]
            base_pos[:, :2] += torch_rand_float(-0.5, 0.5, (len(env_ids), 2), self.device)
        else:
            base_pos = self.simulator.base_init_pos.reshape(1, -1).repeat(len(env_ids), 1)
            base_pos += self.simulator.env_origins[env_ids]

        base_quat = self.simulator.base_init_quat.reshape(1, -1).repeat(len(env_ids), 1)
        base_lin_vel = torch_rand_float(-0.0, 0.0, (len(env_ids), 3), self.device)
        base_ang_vel = torch_rand_float(-0.0, 0.0, (len(env_ids), 3), self.device)

        self.simulator.reset_root_states(env_ids, base_pos, base_quat,
                                          base_lin_vel, base_ang_vel)
```

## 6.7 步骤五：测试新机器人

### 6.7.1 先在调试模式测试

使用少量环境和调试模式：

```bash
export SIMULATOR=genesis  # 或你的模拟器
python legged_gym/scripts/train.py --task myrobot --num_envs 16 --debug
```

### 6.7.2 常见问题排查

#### 问题 1：机器人立即掉落

**可能原因：**
- 初始高度太低
- 默认关节角度不对

**解决方法：**
```python
# 增加初始高度
pos = [0.0, 0.0, 0.5]  # 增加 z 值

# 检查并调整默认关节角度
default_joint_angles = {
    # 调整这些值
}
```

#### 问题 2：关节超出限制

**可能原因：**
- 动作缩放太大
- PD 增益不合适

**解决方法：**
```python
# 减小动作缩放
action_scale = 0.15  # 从 0.25 减小

# 调整 PD 增益
stiffness = {"joint": 15.0}  # 调整刚度
damping = {"joint": 1.0}      # 调整阻尼
```

#### 问题 3：观测索引错误

**可能原因：**
- 观测维度与 `_get_noise_scale_vec` 不匹配

**解决方法：**
```python
# 确保观测维度正确
num_observations = 45  # 调整为实际值

# 确保噪声向量大小匹配
noise_vec = torch.zeros_like(self.obs_buf[0])  # 正确！
```
